class Date
{
private:
	int dd,mm,yyyy;
public:
	Date() 
    {
		dd = 1;
		mm = 1;
		yyyy = 2021;
	}
	Date(int d, int m, int y) 
    {
		dd = d;
		mm = m;
		yyyy = y;
	}
    Date(const Date& temp)
    {
        dd = temp.dd;
        mm = temp.mm;
        yyyy = temp.yyyy;
    }
    void SetDate(int d,int m,int y)
    {
        dd = d;
        mm= m;
        yyyy = y;
    }
    int GetDay()
    {
        return dd;
    }
    int GetMonth()
    {
        return mm;
    }
    int GetYear()
    {
        return yyyy;
    }
    void AddDate()
    {
    while(true){
        cout<<"Enter date (dd mm yyyy) : ";
        cin>>dd>>mm>>yyyy;
        if(yyyy < 1900 || yyyy > 2100) 
        {
            cout<<"Invalid year! Please enter a year between 1900 and 2100."<<endl;
            continue;
        }
        if(mm < 1 || mm > 12) 
        {
            cout<<"Invalid month! Please enter a month between 1 and 12."<<endl;
            continue;
        }
        if(dd < 1 || dd > 31 || (mm == 2 && dd > 29) || ((mm == 4 || mm == 6 || mm == 9 || mm == 11) && dd > 30)) 
        {
            cout<<"Invalid day! Please enter a valid day for the given month."<<endl;
            continue;
        }
        break;
    }
    }
    //void Display(){
    //   cout<<dd<<"/"<<mm<<"/"<<yyyy;
    //}
    string DisplayDateInString()
    {
        stringstream ss;
        ss<<dd<<"/"<<mm<<"/"<<yyyy;
        return ss.str();
    }
    bool Compare(const Date& date)
    {
        if((dd == date.dd) && (mm == date.mm) && (yyyy == date.yyyy))
        {
            return true;
        }
        else
        {
            return false;
        }
    }
};

    {
        for(const auto& ah : accountHolders)
        {
            ah.Display();
        }
    }
    AccountHolder* SearchAccountHolderByID(int id)
    {
        for(auto& ah : accountHolders)
        {
            if(ah.GetHolderID() == id)
            {
                return &ah;
            }
        }
        return nullptr;
    }
    AccountHolder* SearchAccountHolderByName(string name)
    {
        for(auto& ah : accountHolders)
        {
            if(ah.Name == name)
            {
                return &ah;
            }
        }
        return nullptr;
    }
    Person* SearchAccountHolderByUsername(const string& username, const string& password) 
    {
        for (auto& ah : accountHolders) 
        {
            if (ah.GetUsername() == username && ah.GetPassword() == password) 
            {
                return &ah;
            }
        }
        return nullptr;
    }
    bool DeleteAccountHolder(int id)
    {
        for(auto it = accountHolders.begin(); it != accountHolders.end(); ++it)
        {
            if(it->GetHolderID() == id)
            {
                accountHolders.erase(it);
                return true;
            }
        }
        return false;
    }
    void WriteDataToFile(const char* filePath)
    {
        ofstream fileWriter;
        fileWriter.open(filePath);

        if(!fileWriter)
        {
            cout<<"Error opening file!"<<endl;
            return;
        }
        for(const auto& ah : accountHolders)
        {
            fileWriter<<ah.GetHolderID()<<","
                      <<ah.GetUsername()<<","
                      <<ah.GetPassword()<<","
                      <<ah.Name<<","
                      <<ah.Gender<<","
                      <<ah.GetContact()<<","
                      <<ah.GetEmail()<<","
                      <<ah.GetAddress()<<"\n";
        }
        fileWriter.close();
    }
    void ReadDataFromFile(const char* filePath)
    {
        ifstream fileReader;
        fileReader.open(filePath);

        if(!fileReader)
        {
            cout<<"Error opening file!"<<endl;
            return;
        }
        string line;
        while(getline(fileReader, line))
        {
            if(!line.empty())
            {
                AccountHolder ah;
                ah.ReadString(line);
                accountHolders.push_back(ah);
            }
        }
        fileReader.close();
    }

                    cout<<"Enter card number to search : ";
                    string cardNum;
                    cin.ignore();
                    getline(cin,cardNum);
                    BankCard* bankCard = bcollection.SearchBankCardByNumber(cardNum);
                    if(bankCard != nullptr)
                    {
                        bankCard->Display();
                    }
                    else
                    {
                        cout<<"BankCard not found!"<<endl;
                    }
                    break;
                }
                case 2: 
                {
                    cout<<"Enter Card Holder name to search : ";
                    getline(cin,name);
                    BankCard* bankCard = bcollection.SearchBankCardByHolderName(name);
                    if(bankCard != nullptr)
                    {
                        bankCard->Display();
                    }
                    else
                    {
                        cout<<"BankCard not found!"<<endl;
                    }
                    break;
                }
                default:
                SetColour(RED);
                    cout<<"Invalid choice!"<<endl;
                ResetColour();
                break;
            }
            break;
        }
        case 'd':
        {
            cout<<"-------Updating BankCard-------"<<endl;
            cout<<"Enter card number to update : ";
            string cardNum;
            cin.ignore();
            getline(cin,cardNum);
            BankCard* bankCard = bcollection.SearchBankCardByNumber(cardNum);
            if(bankCard != nullptr)
            {
                bankCard->GetBankCard();
                cout<<"BankCard updated successfully!"<<endl;
            }
            else
            {
                cout<<"BankCard not found!"<<endl;
            }
            break;
        }
        case 'e':
        {
            cout<<"-------Deleting BankCard-------"<<endl;
            cout<<"Enter card number to delete : ";
            string cardNum;
            cin>>cardNum;
            if(bcollection.DeleteBankCard(cardNum))
            {
                cout<<"BankCard deleted successfully!"<<endl;
            }
            else
            {
                cout<<"BankCard not found!"<<endl;
            }
            break;
        }
        case 'f':
            cout << "Exiting BankCard Menu" << endl;
            break;
        default:
        SetColour(RED);
            cout<<"Invalid sub-choice"<<endl;
        ResetColour();
        break;
        }
    }while(subChoice != 'f');
}

    int input;
    while (true) {
        cout << "Enter user type (0=UNKNOWN, 1=ADMIN, 2=EMPLOYEE, 3=ACCOUNT_HOLDER): ";
        if (cin >> input) {
            if (input >= 0 && input <= 3) {
                return static_cast<USERTYPE>(input);
            }
        } else {
            cin.clear();
            cin.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
        }
        cout << "Invalid input. Please enter a number between 0-3.\n";
    }
}
